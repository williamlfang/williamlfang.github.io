killx () {
    list=$(ps aux | grep -i $1| grep -v grep |grep -v color)

    if [ -n "$list" ]; then
        dead=$(ps aux | grep -i $1| grep -v grep| grep -v color | awk '{print $2}')
        echo "Killing... $1"
        echo $dead |xargs kill -9
    else
        echo "Not running $1"
    fi
}
