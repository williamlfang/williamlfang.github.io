import os
import openai
openai.api_key = OPEN_API_KEY
completion = openai.ChatCompletion.create(
  model="gpt-3.5-turbo",
  messages=[
    {"role": "user", "content": "show me the futures of programming."}
  ]
)

print(completion.choices[0].message.content)
