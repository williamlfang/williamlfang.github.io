## 查看多少个 cpu 核
cat /proc/cpuinfo |grep "model name"

## 修改 grub
vim /etc/default/grub

## 找到 GRUB_CMDLINE_LINUX 增加
## 隔离 1-2， 7-8
## 从 0 开始计数
ioslcpus=1-2,7-8

## 更新配置
grub2-mkconfig -o /boot/grub2/grub.cfg

## 重启生效
reboot

## 查看是否生效
cat /proc/cmdline
