# 自动完成编译过程

CC      = g++
CFLAGS  = -ggdb3 -std=c++11 -Wall
SRCS    = $(wildcard *.cpp)
PROGS   = $(patsubst %.cpp, %, $(SRCS))
.PHONY: all info clean

all: $(PROGS)

%: %.cpp
	$(CC) $(CFLAGS) -o $@ $<

info:
	@echo "all source files..."
	@echo $(SRCS)
	@echo "all target programms..."
	@echo $(PROGS)

clean:
	rm -f $(PROGS)
