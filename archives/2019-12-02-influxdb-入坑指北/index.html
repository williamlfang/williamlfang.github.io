cd ~/Downloads
## 官网下载 https://portal.influxdata.com/downloads/
wget https://dl.influxdata.com/influxdb/releases/influxdb_1.7.9_amd64.deb

## 开始安装
sudo dpkg -i influxdb_1.7.9_amd64.deb

## 启动服务
/bin/systemctl start influxdb.service
